init(autoreset=True)

url = "secret"
user_key = ""
days_left = 0

def get_hwid():
    raw = str(uuid.getnode()) + os.environ.get("USERNAME", "")
    return hashlib.sha256(raw.encode()).hexdigest()

def authorize():
    global user_key, days_left
    hwid = get_hwid()
    user_key = input("Enter Key: ")
    try:
        r = requests.post(url + "val", json={"key": user_key, "hwid": hwid}, timeout=5)
        data = r.json()
        if data.get("status") == "work":
            days_left = data.get("days_remaining")
            print(Fore.GREEN + f"Authorized | Days left: {days_left}")
            return True
        elif data.get("status") == "hwid_mismatch":
            print(Fore.RED + "Key already bound to another PC!")
        elif data.get("status") == "expired":
            print(Fore.RED + "Key expired!")
        else:
            print(Fore.RED + "Invalid key!")
    except:
        print(Fore.RED + "Authorization server error!")
    return False

if not authorize():
    input("Press Enter to exit...")
    sys.exit()

def cls():
    os.system('cls' if os.name == 'nt' else 'clear')

def check_server():
    try:
        r = requests.get(url, timeout=3)
        if r.status_code == 200 and r.json().get("status") == "work":
            return True
    except:
        pass
    return False

def log_success(func):
    print(Fore.GREEN + f"[Frontal] | {func} Success...")

def log_error(func):
    print(Fore.RED + f"[Frontal] | {func} Error...")

def optimize():
    print(Fore.GREEN + "[Frontal] | Checks...")
    time.sleep(0.5)
    SPI_SETKEYBOARDSPEED = 0x000B
    SPI_SETMOUSESPEED = 0x0071

    try:
        ctypes.windll.user32.SystemParametersInfoW(SPI_SETKEYBOARDSPEED, 31, 0, 0)
        ctypes.windll.user32.SystemParametersInfoW(SPI_SETMOUSESPEED, 0, 20, 0)
        log_success("Keyboard & Mouse Speed")
    except:
        log_error("Keyboard & Mouse Speed")

    try:
        subprocess.run('powercfg /setactive SCHEME_MIN', shell=True)
        log_success("High Performance Power")
    except:
        log_error("High Performance Power")

    try:
        subprocess.run('powershell -Command "Set-ItemProperty -Path HKCU:\\Software\\Microsoft\\GameBar -Name AllowAutoGameMode -Value 1"', shell=True)
        log_success("Game Mode")
    except:
        log_error("Game Mode")

    try:
        key = winreg.OpenKey(winreg.HKEY_CURRENT_USER,
                             r"Software\Microsoft\Windows\CurrentVersion\Explorer\RunMRU",
                             0, winreg.KEY_ALL_ACCESS)
        for i in range(100):
            try:
                winreg.DeleteValue(key, chr(65+i))
            except:
                break
        winreg.CloseKey(key)
        log_success("RunMRU Clean")
    except:
        log_error("RunMRU Clean")

    temp_paths = [os.environ['TEMP'], os.environ['WINDIR'] + "\\Temp"]
    for path in temp_paths:
        try:
            for root, dirs, files in os.walk(path):
                for f in files:
                    try:
                        os.remove(os.path.join(root, f))
                    except:
                        pass
                for d in dirs:
                    try:
                        shutil.rmtree(os.path.join(root, d))
                    except:
                        pass
            log_success(f"Temp Clean: {path}")
        except:
            log_error(f"Temp Clean: {path}")

    try:
        subprocess.run('powershell -Command "Get-Process | Where-Object {$_.Responding -eq $false} | Stop-Process -Force"', shell=True)
        log_success("Kill Not Responding Processes")
    except:
        log_error("Kill Not Responding Processes")
